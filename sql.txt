create table lecture(
id varchar(15) primary key,
name varchar(35),
department varchar(50),
email varchar(50),
password varchar(50));

create table student(
USN varchar(15) primary key,
name varchar(50),
address varchar(50),
phone varchar(10),
gender varchar(7),
dob date,
department varchar(50),
email varchar(50),
password varchar(50));

);
create table stusem(
USN varchar(15) primary key,
subject varchar(50),
marks float,
foreign key(USN) references student(USN) ON delete set NULL,
foreign key(subject) references stusub(subject) ON delete set NULL
);
